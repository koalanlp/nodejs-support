Example
let API = koalanlp.API;
let Tagger = koalanlp.Tagger;
let eunjeonTagger = new Tagger(API.EUNJEON);
let promise = eunjeonTagger.tag("이렇게 사용하면 문단을 분석합니다. 간단하지 않나요?");
// 분석 결과는 Promise 객체에 담겨있습니다.
promise.then(console.log);
